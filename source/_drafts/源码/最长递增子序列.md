function getSequence(arr) {
    const p = arr.slice(); // 反向链表
    const result = [0] // 结果值到原数组索引的映射
    let i, j, u, v, c;
    const len = arr.length;
    for (i = 0; i < len; i++) {
        const arrI = arr[i];
        if (arrI !== 0) {
            j = result[result.length - 1]
            if (arr[j] < arrI) { // 与最后一项对比
                p[i] = j;
                result.push(i);
                continue
            }

            u = 0;
            v = result.length - 1;
            while (u < v) { // 二分查找
                c = (u + v) >> 1;
                if (arr[result[c]] < arrI) {
                    u = c + 1
                } else {
                    v = c
                }
            }
            if (arrI < arr[result[u]]) {
                if (u > 0) {
                    p[i] = result[u - 1]
                }
                result[u] = i
            }
        }
    }

    // 回溯修正
    u = result.length
    v = result[u - 1]
    while (u-- > 0) {
        result[u] = v
        v = p[v]
    }
    return result
}
